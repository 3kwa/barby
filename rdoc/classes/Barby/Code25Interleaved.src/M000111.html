    def encoding_for_pair(pair)
      bars, spaces = ENCODINGS[pair.first], ENCODINGS[pair.last]
      bars.zip(spaces).inject '' do |enc,p|
        bar, space = *p
        enc + ('1' *  (bar == WIDE ? wide_width : narrow_width)) +
        ('0' *  (space == WIDE ? wide_width : narrow_width))
      end
    end
